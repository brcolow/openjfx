  27 import com.sun.glass.ui.*;
  28 import java.nio.IntBuffer;
  29 
  30 final class GtkRobot extends Robot {
  31 
  32     @Override
  33     protected void _create() {
  34         // no-op
  35     }
  36 
  37     @Override
  38     protected void _destroy() {
  39         // no-op
  40     }
  41 
  42     @Override
  43     protected native void _keyPress(int code);
  44 
  45     @Override
  46     protected native void _keyRelease(int code);
  47 
  48     @Override
  49     protected native void _mouseMove(int x, int y);
  50 
  51     @Override
  52     protected native void _mousePress(int buttons);
  53 
  54     @Override
  55     protected native void _mouseRelease(int buttons);
  56 
  57     @Override
  58     protected native void _mouseWheel(int wheelAmt);
  59 
  60     @Override
  61     protected native int _getMouseX();
  62 
  63     @Override
  64     protected native int _getMouseY();
  65 
  66     @Override
  67     protected int _getPixelColor(int x, int y) {
  68         Screen mainScreen = Screen.getMainScreen();
  69         x = (int) Math.floor((x + 0.5) * mainScreen.getPlatformScaleX());
  70         y = (int) Math.floor((y + 0.5) * mainScreen.getPlatformScaleY());
  71         int[] result = new int[1];
  72         _getScreenCapture(x, y, 1, 1, result);
  73         return result[0];
  74     }
  75 
  76     @Override native protected void _getScreenCapture(int x, int y, int width, int height, int[] data);
  77 }
